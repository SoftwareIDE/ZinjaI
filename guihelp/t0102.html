   if (det<0) {
      float p_real = -b/(2*a);
      float p_imag = sqrt(-det)/(2*a);
      cout<<"Las raices son: "<<p_real<<"+"<<p_imag<<"i y "<<p_real<<"-"<<p_imag<<"i"<<endl;
   } else {
      float raiz1 = (-b+sqrt(det))/(2*a);
      float raiz2 = (-b-sqrt(det))/(2*a);
      cout<<"Las raices son: "<<raiz1<<" y "<<raiz2<<endl;
   }
   return 0;
}
